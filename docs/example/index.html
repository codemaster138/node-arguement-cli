API Reference Class: 'Command' Command.option() Command.describe() Command.command() Command.handler() Command.demandArgument() Command.header() Command.usage() Command.noHelp() Command.parse()

Example

A simple example to get you up and running:

const { Command } = require('tauris');

const argv = new Command('example')
  .describe('example command')
  .option('v', {
    alias: [ 'version' ],
    description: 'Display the version number',
    type: 'boolean'
  })
  .demandArgument()
  .parse(process.argv.slice(2));

if (argv) { // `argv` is `false` when a subcommand is invoked
  if (argv.v) console.log('version 1.0.0');
}
